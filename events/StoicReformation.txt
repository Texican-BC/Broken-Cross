		}

		}

			factor = 0.5
			OR = {
				culture_group = byzantine

			}
		}
	}

		name = "Stoicism_embraced_stage1"
		change_religion = Rizana

				name = "conversion_zeal"

		}
		random_owned_province = {
			change_religion = Rizana
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
			if = {
				limit = {
					NOT = { is_capital = yes }
				}
				root = {
					capital_scope = {
						change_religion = Rizana
						add_permanent_province_modifier = {
							name = "religious_zeal_at_conv"

						}
					}
				}
			}
		}
